def call(error)
  cause = error
  case error
  when ContextualError
    cause = error.cause
    @terminal.puts(cause_string(cause))
    @terminal.puts(context_string(error), :bold)
  when ::Interrupt
    @terminal.puts
    @terminal.puts("INTERRUPTED", :bold)
  else
    @terminal.puts(cause_string(error))
  end
  exit_code_for(cause)
end
