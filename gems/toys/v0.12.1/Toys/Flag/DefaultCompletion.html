def call(context)
  results =
    if @include_short && @include_long && @include_negative
      @flag.effective_flags
    else
      collect_results
    end
  fragment = context.fragment
  results.find_all { |val| val.start_with?(fragment) }
         .map { |str| Completion::Candidate.new(str) }
end
