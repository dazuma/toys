def initialize(name: DEFAULT_TOOL_NAME,
               gem_version: nil,
               fail_on_error: true,
               options: [])
  @name = name
  @gem_version = gem_version || DEFAULT_GEM_VERSION_REQUIREMENTS
  @fail_on_error = fail_on_error
  @options = options
end
